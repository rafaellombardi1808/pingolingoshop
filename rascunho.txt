    // ========= HELPERS =========
    const fmtBRL = (v) => v == null ? "Consulte" : v.toLocaleString("pt-BR", {style:"currency", currency:"BRL"});
    const escapeTxt = (s) => (s || "").replace(/[&<>"]/g, ch => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[ch]));

    // ========= UI STATE =========
    const gridEl = document.getElementById("productsGrid");

    // ========= CART (local) =========
    const CART_KEY = "pl_cart_v3";
    let cart = loadCart();

    function loadCart(){
      try{ return JSON.parse(localStorage.getItem(CART_KEY) || "{}"); }
      catch{ return {}; }
    }
    function saveCart(){
      localStorage.setItem(CART_KEY, JSON.stringify(cart));
      updateCartUI();
    }

    function addToCart(id){
      cart[id] = (cart[id] || 0) + 1;
      saveCart();
      openCart();
    }
    function decQty(id){
      if(!cart[id]) return;
      cart[id] -= 1;
      if(cart[id] <= 0) delete cart[id];
      saveCart();
    }
    function incQty(id){
      cart[id] = (cart[id] || 0) + 1;
      saveCart();
    }
    function removeItem(id){
      delete cart[id];
      saveCart();
    }
    function clearCart(){
      cart = {};
      saveCart();
      closeCart();
    }

    function cartCount(){
      return Object.values(cart).reduce((a,b)=>a+b,0);
    }

    function cartTotal(){
      let total = 0;
      for(const [id, qty] of Object.entries(cart)){
        const p = PRODUCTS.find(x=>x.id===id);
        if(!p) continue;
        if(p.price != null) total += p.price * qty;
      }
      return total;
    }

        items.push(`- ${p.name} (x${qty}) ‚Äî ${priceTxt}`);
      }

    }

    // ========= CATALOG RENDER =========
    function getFilteredProducts(){
      const q = (searchInput.value || "").trim().toLowerCase();
      const cat = catSelect.value;

      let list = PRODUCTS.filter(p=>{
        const matchQ =
          !q ||
          p.name.toLowerCase().includes(q) ||
          (p.desc || "").toLowerCase().includes(q) ||
          (p.category || "").toLowerCase().includes(q);

        const matchCat = (cat === "all") || (p.category === cat);
        return matchQ && matchCat;
      });

      const sort = sortSelect.value;
      if(sort === "price-asc"){
        list.sort((a,b)=>(a.price ?? 1e9) - (b.price ?? 1e9));
      }else if(sort === "price-desc"){
        list.sort((a,b)=>(b.price ?? -1) - (a.price ?? -1));
      }else if(sort === "name-asc"){
        list.sort((a,b)=>a.name.localeCompare(b.name, "pt-BR"));
      }
      return list;
    }

    function renderProductMedia(p){

        ? `<img src="${escapeTxt(p.image)}" alt="${escapeTxt(p.name)}" loading="lazy"

        : `<div class="placeholder">SEM FOTO AINDA<br><span style="font-weight:900">coloque em /img</span></div>`;

      return `${badge}${img}`;
    }
